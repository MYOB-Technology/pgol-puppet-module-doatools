  define_environment_resources(
    $name,
    $ensure,
    $region,
    $network,
    $zones,
    $server_roles,
    $services,
    $db_servers,
    $s3,
    $tags,
    $policies,
  ).each |$r| {
    $rt = $r['resource_type']
    $rts = $r['resources'].keys
    info("declaring resources: ${rt} ${rts}")
    debug($r['resources'])
    create_resources($r['resource_type'], $r['resources'], {})
  }
